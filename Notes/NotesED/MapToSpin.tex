\section{Exact Diagonalization Study of Cooling in the Spin Model}

We analyze cooling protocols for fermionic systems and connect them to spin systems via the Jordan-Wigner (JW) transformation. This enables a similar analysis for spin systems, which are more directly implementable in many experimental platforms.

Mapping both the system and bath together presents a challenge: the JW transformation introduces non-local string operators, rendering the combined system non-integrable. Therefore, we apply the transformation only to the system Hamiltonian, which allows us to compute the mode energies $\epsilon_k$ and track which modes are resonating and being cooled during the protocol.

\subsection{The Transverse Field Ising Model}

We study a related model of spins
\begin{align}
    H_{\mathrm{spin}} &= \frac{\cos\theta}{2} \left(\sum_{n=1}^{N-1} \sigma_{n,x}\sigma_{n+1,x} + g_I \sigma_{N,x}\sigma_{1,x}\right) + \frac{\sin\theta}{2} \sum_{n=1}^N \sigma_{n,z}, \label{eq:spin_hamiltonian}
\end{align}
where $\sigma_{n,\alpha}$ ($\alpha = x, y, z$) are the Pauli operators acting on site $n$, and $N$ is the number of spins (assumed to be even). The boundary condition is determined by the parameter $g_I$: $g_I = +1$ for periodic boundary conditions (PBC), $g_I=-1$ for anti-periodic boundary conditions (APBC), and $g_I = 0$ for open boundary conditions (OBC).

\subsection{Jordan-Wigner Transformation}
To map the spin model to fermions, we employ the JW transformation using the string operator $S_n = \prod_{j=1}^{n-1} \sigma_{j,z}$. The fermionic creation and annihilation operators are defined as:
\begin{align}
    a_n      &= -S_n \cdot \frac{\sigma_{n,x} - i\sigma_{n,y}}{2}, \label{eq:JW_a_n}\\
    a_n^\dag &= -S_n \cdot \frac{\sigma_{n,x} + i\sigma_{n,y}}{2}. \label{eq:JW_a_n_dag}
\end{align}
These operators satisfy the standard fermionic anticommutation relations $\{a_m, a_n\} = \{a_m^\dag, a_n^\dag\} = 0$ and $\{a_m, a_n^\dag\} = \delta_{mn}$.

Inverting the transformation, we can express the Pauli operators in terms of fermions:
\begin{align}
    \sigma_{n,x} &= -S_n(a_n + a_n^\dag),\\
    \sigma_{n,y} &= -iS_n(a_n - a_n^\dag),\\
    \sigma_{n,z} &= 2a_n^\dag a_n - 1 = -(a_n a_n^\dag - a_n^\dag a_n).
\end{align}


\subsection{Transforming the Hamiltonian}

We now apply the JW transformation to each term in the Hamiltonian.

For the transverse field term, substituting $\sigma_{n,z} = 2a_n^\dag a_n - 1$, we obtain:
\begin{align}
    \frac{\sin\theta}{2} \sum_{n=1}^N \sigma_{n,z}
     &= \frac{\sin\theta}{2} \sum_{n=1}^N (2a_n^\dag a_n - 1) \nn\\
     &= -\frac{\sin\theta}{2} \sum_{n=1}^N (a_n a_n^\dag - a_n^\dag a_n)
\end{align}


For the nearest-neighbor interaction terms in the bulk, $\sigma_{n,x}\sigma_{n+1,x}$, we start from the fermionic representation of $\sigma_{n,x}$:
\begin{align}
    \sigma_{n,x}\sigma_{n+1,x} &= [-S_n(a_n+a_n^\dag)]\,[-S_{n+1}(a_{n+1}+a_{n+1}^\dag)]\\
                               &= S_n(a_n+a_n^\dag)\,S_{n+1}(a_{n+1}+a_{n+1}^\dag).
\end{align}
Using the identity $S_{n+1}=S_n \sigma_{n,z}$ and $S_n^2=\Id$, we have
\begin{align}
    \sigma_{n,x}\sigma_{n+1,x}
    &= S_n(a_n+a_n^\dag)\,S_n \sigma_{n,z}(a_{n+1}+a_{n+1}^\dag)\nn\\
    &= (a_n+a_n^\dag)\,\sigma_{n,z}(a_{n+1}+a_{n+1}^\dag),
\end{align}
where we used $[S_n,a_n]=0$ and $S_n^2=\Id$.
Using $\sigma_{n,z}=2a_n^\dag a_n - 1$, one verifies the local identity
\begin{align}
    (a_n+a_n^\dag)\sigma_{n,z} = a_n - a_n^\dag.
\end{align}
Therefore,
\begin{align}
    \sigma_{n,x}\sigma_{n+1,x}
    &= (a_n - a_n^\dag)(a_{n+1}+a_{n+1}^\dag)\nn\\
    &= a_n a_{n+1} + a_n a_{n+1}^\dag - a_n^\dag a_{n+1} - a_n^\dag a_{n+1}^\dag.
\end{align}

The boundary term requires special attention due to the non-local nature of the JW strings. For a closed chain ($g_I=\pm 1$), the fermionic Hilbert space splits into two sectors based on fermion parity. The parity operator is defined as
\begin{align}
    P = \prod_{j=1}^{N} \sigma_{j,z} = \prod_{j=1}^{N} (2a_j^\dag a_j - 1) =
    (-1)^{\sum_{j=1}^{N} n_j} = (-1)^{N_f}
\end{align}
where $N_f = \sum_{j=1}^{N} n_j$ is the total fermion number.
For the boundary term, we can show that $S_N S_1 = P \cdot \sigma_{N,z}$, where $P$ is the parity operator (since $S_N=\prod_{j=1}^{N-1} \sigma_{j,z} = \prod_{j=1}^{N} \sigma_{j,z} \cdot \sigma_{N,z}$ and $S_1=\Id$). Therefore:
\begin{align}
    \sigma_{N,x}\sigma_{1,x} &= -P \cdot (a_N - a_N^\dag)(a_1 + a_1^\dag)\\
                             &= -P \cdot (a_N a_1 + a_N a_1^\dag - a_N^\dag a_1 - a_N^\dag a_1^\dag).
\end{align}
In the even parity sector ($P = 1$):
\begin{align}
    \sigma_{N,x}\sigma_{1,x} &= -(a_N a_1 + a_N a_1^\dag - a_N^\dag a_1 - a_N^\dag a_1^\dag).
\end{align}
In the odd parity sector ($P = -1$):
\begin{align}
    \sigma_{N,x}\sigma_{1,x} &= a_N a_1 + a_N a_1^\dag - a_N^\dag a_1 - a_N^\dag a_1^\dag.
\end{align}

Combining all terms, the transformed Hamiltonian in the even/odd parity sector is:
\begin{align}
    H_{\mathrm{spin}}^{e/o} &= \frac{\cos\theta}{2} \left[\sum_{n=1}^{N-1} (a_n - a_n^\dag)(a_{n+1} + a_{n+1}^\dag)
    - g_I P (a_N - a_N^\dag)(a_1 + a_1^\dag)\right] \nn\\
                            &\quad - \frac{\sin\theta}{2} \sum_{n=1}^N (a_n a_n^\dag - a_n^\dag a_n)\\
                            &=\frac{\cos\theta}{2} \left[\sum_{n=1}^{N-1} (a_n a_{n+1} + a_n a_{n+1}^\dag - a_n^\dag a_{n+1} - a_n^\dag a_{n+1}^\dag)
    - g_I P (a_N a_1 + a_N a_1^\dag - a_N^\dag a_1 - a_N^\dag a_1^\dag)\right] \nn\\
                            &\quad - \frac{\sin\theta}{2} \sum_{n=1}^N (a_n a_n^\dag - a_n^\dag a_n)
\end{align}

where $P$ is the fermion parity, which is $+1$ in the even sector and $-1$ in the odd sector.

To emphasize the role of the boundary condition in the fermionic problem, we introduce $g_F = -g_I \cdot P$, which gives:
\begin{align}
    H_{\mathrm{spin}}^{e/o} &= \frac{\cos\theta}{2} \left[\sum_{n=1}^{N-1} (a_n a_{n+1} + a_n a_{n+1}^\dag - a_n^\dag a_{n+1} - a_n^\dag a_{n+1}^\dag) \right. \nn\\
                            &\quad \left. + g_F (a_N a_1 + a_N a_1^\dag - a_N^\dag a_1 - a_N^\dag a_1^\dag)\right] \nn\\
                            &\quad -\frac{\sin\theta}{2} \sum_{n=1}^N (a_n a_n^\dag - a_n^\dag a_n)
\end{align}

Equivalently, in a fixed parity sector (so $g_F$ is a fixed number), we can write the interaction term in a translation-invariant form by defining
the fermionic boundary condition
\begin{align}
    a_{N+1} = g_F a_1,\qquad a_{N+1}^\dag = g_F a_1^\dag.
\end{align}
Then the Hamiltonian can be expressed compactly as
\begin{align}
    H_{\mathrm{spin}}^{g_F}
    &= \frac{\cos\theta}{2}\sum_{n=1}^{N} (a_n-a_n^\dag)(a_{n+1}+a_{n+1}^\dag)
    -\frac{\sin\theta}{2}\sum_{n=1}^{N} (a_n a_n^\dag - a_n^\dag a_n),
    \label{eq:H_ferm_realspace}
\end{align}
where $a_{N+1}$ is understood via the boundary condition above. This is the starting point for the Fourier/Bogoliubov diagonalization below.

% Therefore, in the fermionic problem, it is possible to have antiperiodic boundary conditions or periodic boundary conditions depending on the parity even if the spin model has periodic boundary conditions. 

For a spin system with periodic boundary conditions ($g_I = 1$), the fermionic representation has $g_F = -1$ in the even parity sector and $g_F = +1$ in the odd parity sector, corresponding to antiperiodic and periodic boundary conditions for the fermions, respectively.
For a spin system with anti-periodic boundary conditions ($g_I=-1$), this is swapped: $g_F=+1$ in the even parity sector and $g_F=-1$ in the odd parity sector.

\subsection{Fourier Transformation}
To diagonalize the quadratic fermionic Hamiltonian, we define $\phi_k = \frac{2\pi k}{N}$ and transform to momentum space using the Fourier transform:
\begin{align}
    \ta_k      &= \frac{1}{\sqrt{N}} \sum_{n=1}^N e^{-i n\phi_{k}} a_n,\\
    \ta_k^\dag &= \frac{1}{\sqrt{N}} \sum_{n=1}^N e^{+i n\phi_{k}} a_n^\dag.
\end{align}
The inverse transformations are:
\begin{align}
    a_n      &= \frac{1}{\sqrt{N}} \sum_{k} e^{+i n\phi_k} \ta_k, \label{eq:fourier_def_ann}\\
    a_n^\dag &= \frac{1}{\sqrt{N}} \sum_{k} e^{-i n\phi_k} \ta_k^\dag. \label{eq:fourier_def_cre}
\end{align}
This is the standard ``origin at $n=0$'' Fourier convention written on the site labels $n=1,\dots,N$.
Using $e^{\pm i(n-1)\phi_k}$ instead (common when emphasizing the site index $n=1$ as the origin)
would simply redefine the momentum operators by a $k$-dependent phase
($\ta_k^{(n-1)} = e^{+i\phi_k}\,\ta_k^{(n)}$), and does not affect any physical results as long as the convention
is used consistently.
Here, the sum over $k$ is determined by the fermionic boundary conditions, which depend on the spin parity sector. For periodic boundary conditions ($g_F = 1$), $k$ takes integer values $k \in \{-\frac{N}{2}+1, \dots, \frac{N}{2}\}$. For antiperiodic boundary conditions ($g_F = -1$), $k$ takes half-integer values $k \in \{-\frac{N-1}{2}, \dots, \frac{N-1}{2}\}$.

The Fourier/Bogoliubov diagnostics below are closed-chain transverse-field Ising diagnostics: $N$ is even, $g_I=\pm1$, and a fixed spin-parity sector selects $g_F=\pm1$.  For $g_I=0$, translational invariance is absent and the open-chain BdG construction of \texttt{OBC.tex} must be used instead.  For nonintegrable spin Hamiltonians there is no free-fermion Fourier grid of this form, so the code does not define the \texttt{mode\_hk} and \texttt{mode\_nk} observables.


\subsection{Fermionic PBC $g_F = 1$ Case}

For $g_F=+1$ (fermionic PBC) the allowed momenta are integers
$k\in\{-\frac{N}{2}+1,\dots,\frac{N}{2}\}$ (so $\phi_k=\frac{2\pi k}{N}$ includes the special values $\phi=0$ and $\phi=\pi$).
Using the Fourier transform \cref{eq:fourier_def_ann,eq:fourier_def_cre} and normal ordering, the Hamiltonian decomposes into independent momentum blocks.

It is useful to record the Fourier identities (valid on either momentum grid):
\begin{align}
    \sum_{n=1}^N a_n^\dag a_{n+1} &= \sum_k e^{+i\phi_k}\,\ta_k^\dag\ta_k, \qquad
    \sum_{n=1}^N a_n a_{n+1} = \sum_k e^{-i\phi_k}\,\ta_k\ta_{-k},\\
    \sum_{n=1}^N a_n^\dag a_{n+1}^\dag &= \sum_k e^{+i\phi_k}\,\ta_k^\dag\ta_{-k}^\dag,\qquad
    \sum_{n=1}^N a_n a_{n+1}^\dag = \sum_k e^{-i\phi_k}\,\ta_k\ta_k^\dag.
\end{align}
For completeness, we show the derivation of the first identity (the rest follow analogously).
Using \cref{eq:fourier_def_ann,eq:fourier_def_cre} and the boundary-compatible momentum grid,
\begin{align}
    \sum_{n=1}^N a_n^\dag a_{n+1}
    &= \frac{1}{N}\sum_{n=1}^N \sum_{k,k'} e^{-in\phi_k} e^{+i(n+1)\phi_{k'}}\,\ta_k^\dag \ta_{k'} \nn\\
    &= \frac{1}{N}\sum_{k,k'} e^{+i\phi_{k'}} \left(\sum_{n=1}^N e^{in(\phi_{k'}-\phi_k)}\right)\ta_k^\dag \ta_{k'} \nn\\
    &= \sum_k e^{+i\phi_k}\,\ta_k^\dag \ta_k,
\end{align}
where we used $\sum_{n=1}^N e^{in(\phi_{k'}-\phi_k)}=N\delta_{k,k'}$ on the allowed momentum grid.
As a second example (relevant for the pairing sector), we derive
$\sum_n a_n a_{n+1}=\sum_k e^{-i\phi_k}\ta_k\ta_{-k}$:
\begin{align}
    \sum_{n=1}^N a_n a_{n+1}
    &= \frac{1}{N}\sum_{n=1}^N \sum_{k,k'} e^{+in\phi_k} e^{+i(n+1)\phi_{k'}}\,\ta_k \ta_{k'} \nn\\
    &= \frac{1}{N}\sum_{k,k'} e^{+i\phi_{k'}} \left(\sum_{n=1}^N e^{in(\phi_{k}+\phi_{k'})}\right)\ta_k \ta_{k'} \nn\\
    &= \sum_k e^{-i\phi_{k}}\,\ta_k \ta_{-k},
\end{align}
where we used $\sum_{n=1}^N e^{in(\phi_{k}+\phi_{k'})}=N\delta_{k',-k}$ and $\phi_{-k}=-\phi_k$.

\paragraph{Deriving the $(k,-k)$ block.}
We now show the intermediate steps leading from the real-space fermion Hamiltonian \cref{eq:H_ferm_realspace}
to the canonical paired-mode form \cref{eq:H_pair_form}.
First expand the XX part and use the Fourier identities above:
\begin{widetext}
    \begin{align}
        H_{\mathrm{spin}}^{g_F=1}
        &= \frac{\cos\theta}{2}\sum_{n=1}^{N}(a_n-a_n^\dag)(a_{n+1}+a_{n+1}^\dag)
           -\frac{\sin\theta}{2}\sum_{n=1}^{N}(a_n a_n^\dag-a_n^\dag a_n)\nn\\
        &= \frac{\cos\theta}{2}\sum_{n=1}^{N}\Big(a_n a_{n+1} + a_n a_{n+1}^\dag - a_n^\dag a_{n+1} - a_n^\dag a_{n+1}^\dag\Big)
           -\frac{\sin\theta}{2}\sum_{n=1}^{N}(a_n a_n^\dag-a_n^\dag a_n)\nn\\
        &= \frac{\cos\theta}{2}\sum_{k}\Big(e^{-i\phi_k}\ta_k\ta_{-k}
           + e^{-i\phi_k}\ta_k\ta_k^\dag
           - e^{+i\phi_k}\ta_k^\dag\ta_k
           - e^{+i\phi_k}\ta_k^\dag\ta_{-k}^\dag\Big)
           -\frac{\sin\theta}{2}\sum_{k}\Big(\ta_k\ta_k^\dag-\ta_k^\dag\ta_k\Big).
        \label{eq:H_momentum_raw}
    \end{align}
\end{widetext}
Now consider a generic paired mode $k\neq 0,N/2$ and group the $k$ and $-k$ contributions (using $\phi_{-k}=-\phi_k$ on either momentum grid).
It is convenient to separate number and pairing sectors.
\begin{align}
    \mathcal{N}_{k,-k}
    &\equiv e^{-i\phi_k}\ta_k\ta_k^\dag - e^{+i\phi_k}\ta_k^\dag\ta_k
      \nn\\
    &\quad
      + e^{+i\phi_k}\ta_{-k}\ta_{-k}^\dag - e^{-i\phi_k}\ta_{-k}^\dag\ta_{-k}\nn\\
    &= (e^{-i\phi_k}+e^{+i\phi_k})
       - (e^{-i\phi_k}+e^{+i\phi_k})\,\ta_k^\dag\ta_k\nn\\
    &\quad
       - (e^{-i\phi_k}+e^{+i\phi_k})\,\ta_{-k}^\dag\ta_{-k}\nn\\
    &= -2\cos\phi_k\,(\ta_k^\dag\ta_k+\ta_{-k}^\dag\ta_{-k}-1),
\end{align}
where we used normal ordering $\ta_k\ta_k^\dag=1-\ta_k^\dag\ta_k$ and $e^{-i\phi}+e^{+i\phi}=2\cos\phi$.
Similarly for the pairing sector,
\begin{align}
    \mathcal{P}_{k,-k}
    &\equiv e^{-i\phi_k}\ta_k\ta_{-k} - e^{+i\phi_k}\ta_k^\dag\ta_{-k}^\dag
      \nn\\
    &\quad
      + e^{+i\phi_k}\ta_{-k}\ta_k - e^{-i\phi_k}\ta_{-k}^\dag\ta_k^\dag\nn\\
    &= (e^{-i\phi_k}-e^{+i\phi_k})\,(\ta_k\ta_{-k}+\ta_k^\dag\ta_{-k}^\dag)\nn\\
    &= -2i\sin\phi_k\,(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k),
\end{align}
where we used fermionic anticommutation to reorder $\ta_{-k}\ta_k=-\ta_k\ta_{-k}$ and $\ta_{-k}^\dag\ta_k^\dag=-\ta_k^\dag\ta_{-k}^\dag$,
and then Euler's identity $e^{-i\phi}-e^{+i\phi}=-2i\sin\phi$.

Combining the $k$ and $-k$ pieces of \cref{eq:H_momentum_raw} gives
\begin{align}
    H_{k,-k}
    &= \frac{\cos\theta}{2}\big(\mathcal{N}_{k,-k}+\mathcal{P}_{k,-k}\big)\nn\\
    &\quad
       -\frac{\sin\theta}{2}\Big[(\ta_k\ta_k^\dag-\ta_k^\dag\ta_k)
       +(\ta_{-k}\ta_{-k}^\dag-\ta_{-k}^\dag\ta_{-k})\Big]\nn\\
    &= \big(\sin\theta-\cos\theta\cos\phi_k\big)\,(\ta_k^\dag\ta_k+\ta_{-k}^\dag\ta_{-k}-1)\nn\\
    &\quad
       + i\big(-\cos\theta\sin\phi_k\big)\,(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k),
\end{align}
which is the canonical form quoted below.

Applying these steps to \cref{eq:H_ferm_realspace} and normal ordering $\ta_k\ta_k^\dag=1-\ta_k^\dag\ta_k$, the $(k,-k)$ contribution for a generic paired mode ($k\neq 0,N/2$) can be written as
\begin{align}
    H_{k,-k}
    &= w_k\,(\ta_k^\dag\ta_k+\ta_{-k}^\dag\ta_{-k}-1)
       + i r_k\,(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k),
    \label{eq:H_pair_form}
\end{align}
with
\begin{align}
    w_k = \sin\theta - \cos\theta\cos\phi_k,\qquad
    r_k = -\cos\theta\sin\phi_k.
\end{align}

For the generic paired modes $k\neq 0, N/2$ we define the Nambu spinor
$\tal_k=(\ta_k,\ta_{-k}^\dag)^T$ and obtain
\begin{align}
    H_{\mathrm{spin}}^{g_F=1}
    &= \sum_{k=1}^{N/2-1} \tal_k^\dag \tH_k \tal_k + H_{k=0} + H_{k=N/2},
\end{align}
where the $2\times 2$ Bogoliubov--de Gennes block is
\begin{align}
    \tH_k &=
    \begin{pmatrix}
        w_k   & i r_k\\
        -i r_k& -w_k
    \end{pmatrix},
    \qquad
    w_k = \sin\theta - \cos\theta\cos\phi_k,\qquad
    r_k = -\cos\theta\sin\phi_k.
    \label{eq:bdg_block}
\end{align}
As a quick consistency check, expanding $\tal_k^\dag\tH_k\tal_k$ gives
\begin{align}
    \tal_k^\dag \tH_k \tal_k
    &= w_k(\ta_k^\dag\ta_k-\ta_{-k}\ta_{-k}^\dag)
       + i r_k(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k)\nn\\
    &= w_k(\ta_k^\dag\ta_k+\ta_{-k}^\dag\ta_{-k}-1)
       + i r_k(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k),
\end{align}
which matches \cref{eq:H_pair_form} after normal ordering.
The special modes have $r_k=0$ and are already diagonal:
\begin{align}
    H_{k=0} &= \frac{w_0}{2}(\ta_0^\dag\ta_0 - \ta_0\ta_0^\dag),\qquad w_0=\sin\theta-\cos\theta,\\
    H_{k=N/2} &= \frac{w_{N/2}}{2}(\ta_{N/2}^\dag\ta_{N/2} - \ta_{N/2}\ta_{N/2}^\dag),\qquad w_{N/2}=\sin\theta+\cos\theta.
\end{align}

\subsection{Fermionic APBC $g_F = -1$ Case}

For $g_F=-1$ (fermionic APBC) the allowed momenta are half-integers
$k\in\{-\frac{N-1}{2},\dots,\frac{N-1}{2}\}$.
There are no special modes ($\phi_k\not\in\{0,\pi\}$), and the Hamiltonian again decomposes into independent Nambu blocks of the form \cref{eq:bdg_block}:
\begin{align}
    H_{\mathrm{spin}}^{g_F=-1} &= \sum_{k=1/2}^{(N-1)/2} \tal_k^\dag \tH_k \tal_k.
\end{align}


\section{Bogoliubov Transformation}
\label{sec:bogoliubov}
% NOTE (2026-02-04): Bogoliubov conventions are aligned with the numerical
% implementation used for ED diagnostics (see Obs/ComputeModeBogoliubovData.m).
% With the Fourier transform \cref{eq:fourier_def_ann,eq:fourier_def_cre}, the
% BdG block \cref{eq:bdg_block} has
%   w_k = \sin\theta - \cos\theta\cos\phi_k,
%   r_k = -\cos\theta\sin\phi_k,
%   \epsilon_k = \sqrt{w_k^2+r_k^2} = \sqrt{1-\sin(2\theta)\cos\phi_k},
% and the special-mode labels are fixed by \phi_0=0 and \phi_{N/2}=\pi so that
% w_0=\sin\theta-\cos\theta and w_{N/2}=\sin\theta+\cos\theta. No additional
% gauge shift is applied.
To fully diagonalize the Hamiltonian and obtain the free quasiparticle energies, we apply a Bogoliubov transformation. For modes with $k \neq 0, N/2$, we need to diagonalize the $2 \times 2$ matrix $\tH_k$. The Bogoliubov transformation mixes the $\ta_k$ and $\ta_{-k}^\dag$ operators:
\begin{align}
    \hat{a}_k         &= \cos(\varphi_k) \ta_k + i\sin(\varphi_k) \ta_{-k}^\dag,\\
    \hat{a}_{-k}^\dag &= i\sin(\varphi_k) \ta_k + \cos(\varphi_k) \ta_{-k}^\dag,
\end{align}
where the Bogoliubov angle $\varphi_k$ is chosen to diagonalize $\tH_k$.

To find the angle $\varphi_k$, we need to diagonalize the matrix:
\begin{align}
    \tH_k &=
    \begin{pmatrix}
        w_k   &ir_k\\
        -ir_k &-w_k
    \end{pmatrix}.
\end{align}
The eigenvalues of this matrix are $\pm\epsilon_k$ where:
\begin{align}
    \epsilon_k &= \sqrt{w_k^2 + r_k^2}\\
               &= \sqrt{(\sin\theta - \cos\theta\cos\phi_k)^2 + (\cos\theta\sin\phi_k)^2}\nn\\
               &= \sqrt{\sin^2\theta - 2\sin\theta\cos\theta\cos\phi_k + \cos^2\theta(\cos^2\phi_k+\sin^2\phi_k)}\nn\\
               &= \sqrt{1 - 2\sin\theta\cos\theta\cos\phi_k}
               = \sqrt{1 - \sin(2\theta)\cos\phi_k}.
\end{align}
The Bogoliubov angle is determined by:
\begin{align}
    \tan(2\varphi_k) &= \frac{r_k}{w_k}\\
                     &= \frac{-\cos\theta\sin\phi_k}{\sin\theta - \cos\theta\cos\phi_k}.
\end{align}

Let us verify that this transformation diagonalizes the Hamiltonian. The transformation can be written in matrix form as:
\begin{align}
    \begin{pmatrix}
        \hat{a}_k\\
        \hat{a}_{-k}^\dag
    \end{pmatrix}
    = U_k
    \begin{pmatrix}
        \ta_k\\
        \ta_{-k}^\dag
    \end{pmatrix},
    \quad
    U_k =
    \begin{pmatrix}
        \cos\varphi_k  & i\sin\varphi_k\\
        i\sin\varphi_k &\cos\varphi_k
    \end{pmatrix}.
\end{align}
The Hamiltonian matrix for mode $k$ is diagonalized by this transformation.
With the correct choice of $\varphi_k$, we have:
\begin{align}
    U_k \tH_k U_k^\dag =
    \begin{pmatrix}
        \epsilon_k &0\\
        0          &-\epsilon_k
    \end{pmatrix}.
\end{align}

Therefore, for $k \in \{1, \ldots, N/2-1\}$:
\begin{align}
    \tal_k^\dag \tH_k \tal_k &= \epsilon_k \hat{a}_k^\dag \hat{a}_k - \epsilon_k \hat{a}_{-k} \hat{a}_{-k}^\dag\\
                             &= \epsilon_k \hat{a}_k^\dag \hat{a}_k - \epsilon_k (1 - \hat{a}_{-k}^\dag \hat{a}_{-k})\\
                             &= \epsilon_k (\hat{a}_k^\dag \hat{a}_k - \hat{a}_{-k}\hat{a}_{-k}^\dag).
\end{align}

For the special modes $k=0$ and $k=N/2$ (present only for fermionic PBC, $g_F=+1$), we have $\sin\phi_k=0$ so $r_k=0$ and the Hamiltonian is already diagonal:
\begin{align}
    H_{k=0}
     &= \frac{w_0}{2}(\ta_0^\dag \ta_0 - \ta_0 \ta_0^\dag)
      = \frac{w_0}{2}(\hat{a}_0^\dag \hat{a}_0 - \hat{a}_0 \hat{a}_0^\dag),\\
    H_{k=N/2}
     &= \frac{w_{N/2}}{2}(\ta_{N/2}^\dag \ta_{N/2} - \ta_{N/2} \ta_{N/2}^\dag)
      = \frac{w_{N/2}}{2}(\hat{a}_{N/2}^\dag \hat{a}_{N/2} - \hat{a}_{N/2} \hat{a}_{N/2}^\dag),
\end{align}
with $w_0=\sin\theta-\cos\theta$ and $w_{N/2}=\sin\theta+\cos\theta$. The corresponding excitation energies are $\epsilon_0=|w_0|$ and $\epsilon_{N/2}=|w_{N/2}|$.

Collecting all modes, define the Bogoliubov number operator
\begin{align}
    \hat n_k \equiv \hat a_k^\dag \hat a_k,\qquad
    h_k \equiv 2\hat n_k-1.
    \label{eq:hk_individual_def}
\end{align}
For a generic paired block, \cref{eq:H_pair_form} becomes
\begin{align}
    \tal_k^\dag \tH_k \tal_k
    = \epsilon_k(\hat n_k+\hat n_{-k}-1)
    = \frac{\epsilon_k}{2}(h_k+h_{-k}).
\end{align}
For a special mode $q\in\{0,N/2\}$, present only when $g_F=+1$, the contribution is instead
\begin{align}
    H_q=\frac{w_q}{2}h_q,
\end{align}
with the signed coefficient $w_q$ rather than $|w_q|$.
Thus the full-grid form used by the numerical diagnostics is
\begin{align}
    H_{\mathrm{spin}}^{g_F}
    = \frac{1}{2}\sum_{k\in\mathrm{grid}(g_F)} \operatorname{coeff}_k\,h_k,
    \label{eq:H_full_grid_hk}
\end{align}
where
\begin{align}
    \operatorname{coeff}_k =
    \begin{cases}
        \epsilon_k, & \sin\phi_k\neq 0,\\
        w_k,        & \sin\phi_k=0.
    \end{cases}
\end{align}
In the code parametrization
$H_{\mathrm{code}}=J\sum_j\sigma^z_j\sigma^z_{j+1}+h\sum_j\sigma^x_j$,
related to this note by the global rotation $R_y(\pi/2)^{\otimes N}$, the energy scale is
$\Lambda=2\sqrt{J^2+h^2}$ with $\theta=\operatorname{atan}(h,J)$, and the reconstructed energy is
\begin{align}
    \langle H_{\mathrm{code}}\rangle
    = \frac{\Lambda}{2}\sum_{k\in\mathrm{grid}(g_F)}
      \operatorname{coeff}_k\,\langle h_k\rangle.
    \label{eq:code_energy_from_hk}
\end{align}
The Bogoliubov quasiparticle occupation reported in the cooling diagnostics is therefore
\begin{align}
    n_k^{\mathrm{Bog}} = \langle \hat n_k\rangle
    = \frac{1+\langle h_k\rangle}{2}.
    \label{eq:bog_occupation_from_hk}
\end{align}
This occupation should not be confused with the raw Fourier occupation
\begin{align}
    \tilde n_k
    \equiv \langle \ta_k^\dag\ta_k\rangle
    = \frac{1}{N}\sum_{m,n=1}^{N}e^{i(m-n)\phi_k}
      \langle a_m^\dag a_n\rangle .
    \label{eq:raw_fourier_occupation}
\end{align}
With the relabeled Fourier sign used in the code one instead writes
\(\tilde n_k^{\mathrm{code}}=\tilde n_{-k}\), equivalently replacing
\(e^{i(m-n)\phi_k}\) in \cref{eq:raw_fourier_occupation} by
\(e^{i(n-m)\phi_k}\).  This is only a relabeling of the momentum grid, but it
must not be hidden in the notation.  In the chosen-operator vacuum,
\(\tilde n_k=\sin^2\varphi_k\) for generic modes, whereas
\(n_k^{\mathrm{Bog}}=0\) and \(\langle h_k\rangle=-1\).
The separate dashed reference used in the raw-Fourier plots is the
parity-unconstrained, mode-wise energy-minimizing BdG reference:
for generic modes it is again \(\sin^2\varphi_k\), while for a special mode
it has \(\tilde n_k^{\mathrm{ref}}=1\) if \(w_k<0\),
\(\tilde n_k^{\mathrm{ref}}=0\) if \(w_k>0\), and \(1/2\) at exact
degeneracy.  This reference is not, in general, the fixed-parity sector
ground-state occupation.
The energy reconstruction uses \(h_k\) and the signed coefficient
\(\operatorname{coeff}_k\), not the expression \(\epsilon_k\tilde n_k\).
The individual observables \(h_k\) and the signed coefficients
\(\operatorname{coeff}_k\) in \cref{eq:code_energy_from_hk} are the defining
convention for mode-resolved energy diagnostics. The ED and tensor-network
implementations are tested against the same spin-state expectation values of
these \(h_k\)'s, with the special-mode signs kept in
\(\operatorname{coeff}_k\) rather than absorbed into an absolute excitation
energy. The separate density-matrix cooling-channel convention is recorded in
the tensor-network note; it does not change the Jordan--Wigner formulas above.

\subsection{Going back to the spin operators}

To track the cooling of specific modes in the spin model, the ED and tensor-network diagnostics measure the expectation value of the individual mode operators $h_k$ in \cref{eq:hk_individual_def}. Our goal is to express these operators in terms of the original Pauli operators, which involves tracing back through the sequence of transformations (Bogoliubov, Fourier, JW).

\subsubsection{Generic modes ($k \neq 0, N/2$)}

It is useful first to record the paired-block observable
\begin{align}
    \mathfrak h_{k,-k}\equiv \frac{h_k+h_{-k}}{2}
    =\hat n_k+\hat n_{-k}-1.
\end{align}
Writing
\begin{align}
    \begin{pmatrix}
        \hat{a}_k\\
        \hat{a}_{-k}^\dag
    \end{pmatrix}
    = U_k
    \begin{pmatrix}
        \ta_k\\
        \ta_{-k}^\dag
    \end{pmatrix},
    \qquad
    U_k =
    \begin{pmatrix}
        \cos\varphi_k  & i\sin\varphi_k\\
        i\sin\varphi_k & \cos\varphi_k
    \end{pmatrix},
\end{align}
and using the fermionic anticommutation relations, one obtains the symmetric paired-block form
\begin{align}
    \mathfrak h_{k,-k}
    &= \cos(2\varphi_k)(\ta_k^\dag\ta_k + \ta_{-k}^\dag\ta_{-k} - 1)\nn\\
    &\quad
      + \sin(2\varphi_k)\,
        i(\ta_k^\dag\ta_{-k}^\dag - \ta_{-k}\ta_k).
    \label{eq:hk_bogo_symmetric}
\end{align}
The individual diagnostic stored by the code is instead
\begin{align}
    h_k
    &=2\hat a_k^\dag\hat a_k-1\nn\\
    &=2\cos^2\varphi_k\,\ta_k^\dag\ta_k
      +2\sin^2\varphi_k\,(1-\ta_{-k}^\dag\ta_{-k})\nn\\
    &\quad
      +2\sin\varphi_k\cos\varphi_k\,
       i(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k)-1.
    \label{eq:hk_individual_momentum}
\end{align}
Now, we substitute the Fourier transforms to express $h_k$ in terms of real-space operators $a_n$:
\begin{align}
    \ta_k^\dag \ta_k         &= \frac{1}{N} \sum_{n,m=1}^N e^{i (n-m)\phi_k} a_n^\dag a_m,\\
    \ta_{-k}^\dag \ta_{-k}   &= \frac{1}{N} \sum_{n,m=1}^N e^{i (m-n)\phi_k} a_n^\dag a_m,\\
    \ta_k^\dag \ta_{-k}^\dag &= \frac{1}{N} \sum_{n,m=1}^N e^{i (n-m)\phi_k} a_n^\dag a_m^\dag,\\
    \ta_{-k} \ta_k           &= \frac{1}{N} \sum_{n,m=1}^N e^{i (n-m)\phi_k} a_n a_m.
\end{align}


% The fermionic bilinears become two-spin operators dressed by JW strings. To avoid ambiguity in operator ordering, we always write the strings in the split form $S_n\,\sigma_{n,\alpha}\,S_m\,\sigma_{m,\beta}$ (and not $S_n S_m\,\sigma_{n,\alpha} \sigma_{m,\beta}$). Using $\sigma_n^\pm = (\sigma_{n,x} \pm i\sigma_{n,y})/2$ and $a_n = -S_n \sigma_n^-$, we have:

We next rewrite the JW bilinears in terms of split-string spin operators. Using $\sigma_n^\pm = (\sigma_{n,x} \pm i\sigma_{n,y})/2$ and $a_n = -S_n \sigma_n^-$, we obtain the identities
\begin{widetext}
    \begin{align}
        a_n^\dag a_m      &= S_n \sigma_n^+ S_m \sigma_m^- = \frac{1}{4} S_n (\sigma_{n,x}+i\sigma_{n,y}) S_m (\sigma_{m,x}-i\sigma_{m,y})\nn\\
                          &= \frac{1}{4}\Big( S_n \sigma_{n,x} S_m \sigma_{m,x} + S_n \sigma_{n,y} S_m \sigma_{m,y}
        + i(S_n \sigma_{n,y} S_m \sigma_{m,x} - S_n \sigma_{n,x} S_m \sigma_{m,y}) \Big),
        \label{eq:JW_bilin_1}\\
        a_n^\dag a_m^\dag &= S_n \sigma_n^+ S_m \sigma_m^+ = \frac{1}{4} S_n (\sigma_{n,x}+i\sigma_{n,y}) S_m (\sigma_{m,x}+i\sigma_{m,y})\nn\\
                          &= \frac{1}{4}\Big( S_n \sigma_{n,x} S_m \sigma_{m,x} - S_n \sigma_{n,y} S_m \sigma_{m,y}
        + i(S_n \sigma_{n,x} S_m \sigma_{m,y} + S_n \sigma_{n,y} S_m \sigma_{m,x}) \Big),
        \label{eq:JW_bilin_2}\\
        a_n a_m           &= S_n \sigma_n^- S_m \sigma_m^- = \frac{1}{4} S_n (\sigma_{n,x}-i\sigma_{n,y}) S_m (\sigma_{m,x}-i\sigma_{m,y})\nn\\
                          &= \frac{1}{4}\Big( S_n \sigma_{n,x} S_m \sigma_{m,x} - S_n \sigma_{n,y} S_m \sigma_{m,y}
        - i(S_n \sigma_{n,x} S_m \sigma_{m,y} + S_n \sigma_{n,y} S_m \sigma_{m,x}) \Big).
        \label{eq:JW_bilin_3}
    \end{align}
\end{widetext}
For compactness in what follows, we denote the split-string two-point spin correlators by
\begin{align}
    C^{\alpha\beta}_{nm} \equiv \big\langle S_n \sigma_{n,\alpha} S_m \sigma_{m,\beta} \big\rangle , \qquad \alpha,\beta\in\{x,y\}.
\end{align}

Third, we compute the individual diagnostic $\langle h_k\rangle$ by combining density and pairing sectors. To this end, define
\begin{align}
    u \equiv \cos\varphi_k,\qquad v\equiv\sin\varphi_k.
\end{align}
From \cref{eq:hk_individual_momentum}, we have
\begin{align}
    \langle h_k\rangle
    &= 2u^2\langle\ta_k^\dag\ta_k\rangle
    +2v^2\Big(1-\langle\ta_{-k}^\dag\ta_{-k}\rangle\Big)\nn\\
    &\quad
    +2uv\Big\langle i(\ta_k^\dag\ta_{-k}^\dag-\ta_{-k}\ta_k)\Big\rangle-1.
    \label{eq:hk_expect_split}
\end{align}

We now evaluate the three expectation values in \cref{eq:hk_expect_split} explicitly.
Introduce the shorthand $\theta_{nm}=(n-m)\phi_k$.

\paragraph{Density sector.}
Using the Fourier relations in \cref{eq:fourier_def_ann,eq:fourier_def_cre}, we have
\begin{align}
    \ta_k^\dag\ta_k
    &= \frac{1}{N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,a_n^\dag a_m,\\
    \ta_{-k}^\dag\ta_{-k}
    &= \frac{1}{N}\sum_{n,m=1}^N e^{-i\theta_{nm}}\,a_n^\dag a_m.
\end{align}
These two terms must be kept separately for the individual observable $h_k$; only the paired-block observable $\mathfrak h_{k,-k}$ combines them into a cosine.

\paragraph{Pairing sector.}
Similarly,
\begin{align}
    \ta_k^\dag\ta_{-k}^\dag
    &= \frac{1}{N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,a_n^\dag a_m^\dag,\\
    \ta_{-k}\ta_k
    &= \frac{1}{N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,a_n a_m,
\end{align}
so that
\begin{align}
    \Big\langle i(\ta_k^\dag\ta_{-k}^\dag - \ta_{-k}\ta_k) \Big\rangle
    &= \frac{i}{N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,\langle a_n^\dag a_m^\dag - a_n a_m\rangle.
\end{align}

We next expand the real-space fermion bilinears using the split-string correlators
$C^{\alpha\beta}_{nm}=\langle S_n\sigma_{n,\alpha}S_m\sigma_{m,\beta}\rangle$.
From \cref{eq:JW_bilin_1,eq:JW_bilin_2,eq:JW_bilin_3}, we have
\begin{align}
    \langle a_n^\dag a_m\rangle
    &= \frac{1}{4}\left(C^{xx}_{nm}+C^{yy}_{nm}+ i\,[C^{yx}_{nm}-C^{xy}_{nm}]\right),\\
    \langle a_n^\dag a_m^\dag - a_n a_m\rangle
    &= \frac{i}{2}\left(C^{xy}_{nm}+C^{yx}_{nm}\right).
\end{align}
The second identity follows by subtracting \cref{eq:JW_bilin_3} from \cref{eq:JW_bilin_2}.

As a useful diagonal check, for $n=m$ we have $S_n^2=\Id$ and $\sigma_{n,\alpha}^2=\Id$, hence
$C^{xx}_{nn}=C^{yy}_{nn}=1$, while the Pauli algebra gives
$i[C^{yx}_{nn}-C^{xy}_{nn}]=2\langle\sigma_{n,z}\rangle$.
Thus $\langle a_n^\dag a_n\rangle=\frac{1}{2}(1+\langle\sigma_{n,z}\rangle)$ is recovered automatically from the compact correlator combination above.

Combining these identities yields the explicit result
\begin{widetext}
    \begin{align}
        \langle h_k \rangle
        &= \frac{u^2}{2N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,
           \left(C^{xx}_{nm}+C^{yy}_{nm}+ i\,[C^{yx}_{nm}-C^{xy}_{nm}]\right)\nn\\
        &\quad
        - \frac{v^2}{2N}\sum_{n,m=1}^N e^{-i\theta_{nm}}\,
           \left(C^{xx}_{nm}+C^{yy}_{nm}+ i\,[C^{yx}_{nm}-C^{xy}_{nm}]\right)\nn\\
        &\quad
        - \frac{uv}{N}\sum_{n,m=1}^N e^{i\theta_{nm}}\,
           \left(C^{xy}_{nm}+C^{yx}_{nm}\right)
        +2v^2-1.
        \label{eq:hk_final}
    \end{align}
\end{widetext}
The right-hand side is real, since $h_k$ is Hermitian. Any nonzero imaginary part provides a useful numerical check of correlator conventions and implementations. Note that the mixed correlators $C^{xy}$ and $C^{yx}$ are generally non-Hermitian, so they must be evaluated with the split-string order as written for each pair $(n,m)$ (do not fill them by conjugate mirroring unless one also swaps operators appropriately). The ED implementation uses the opposite Fourier sign for the intermediate operator $\ta_k$, which is the relabeling $k\mapsto -k$; the tensor-network implementation uses the same relabeled phases and the same individual observable \cref{eq:hk_individual_def}.

\subsubsection{Special modes ($k=0, N/2$)}

Special modes occur only for fermionic PBC ($g_F=+1$), where the momentum grid contains $k=0$ and $k=N/2$ (i.e., $\phi_k\in\{0,\pi\}$).
For these modes $\sin\phi_k=0$ so $r_k=0$ and there is no Bogoliubov mixing; we set $\varphi_k=0$ and identify $\hat{a}_k=\ta_k$.
The Hamiltonian contribution is
\begin{align}
    H_k = \frac{w_k}{2}\,h_k,\qquad h_k = 2\ta_k^\dag \ta_k - 1,
\end{align}
with $w_0=\sin\theta-\cos\theta$ and $w_{N/2}=\sin\theta+\cos\theta$ and excitation energies $\epsilon_k=|w_k|$.
In a fixed fermion-parity sector, the special-mode occupations are constrained. In any sector where a single special-mode occupation is forbidden, that mode can remain at $\langle h_k\rangle=-1$ even when $w_0$ changes sign at $\theta=\pi/4$; the sign change is then carried by the \emph{signed} coefficient $w_k$ in \cref{eq:H_full_grid_hk}, not by replacing $w_k$ with $|w_k|$.

In terms of split-string correlators, one may obtain $\langle h_k\rangle$ by taking $v=0$ in \cref{eq:hk_final}.
Since $e^{i\theta_{nm}}=\cos\theta_{nm}$ for $\phi_k\in\{0,\pi\}$, this reduces to
\begin{align}
    \langle h_k\rangle
    &= \frac{1}{2N}\sum_{n,m=1}^N
       \cos\theta_{nm}\left(C^{xx}_{nm}+C^{yy}_{nm}\right)\nn\\
    &\quad
      + \frac{i}{2N}\sum_{n,m=1}^N
        \cos\theta_{nm}\left(C^{yx}_{nm}-C^{xy}_{nm}\right) - 1,
\end{align}
where $\theta_{nm}=(n-m)\phi_k$.

\subsection{Ground State Parity and Gap Structure}

The parity operator $P = \prod_{j=1}^{N} \sigma_{j,z}$ commutes with $H_{\mathrm{spin}}$, splitting the Hilbert space into even ($P = +1$) and odd ($P = -1$) sectors. From the mapping $g_F = -g_I \cdot P$, the spin PBC ($g_I = 1$) yields fermionic APBC ($g_F = -1$) in the even sector and fermionic PBC ($g_F = +1$) in the odd sector.

More generally, in a fixed parity sector $P=\pm 1$ the fermionic boundary condition is $g_F=-g_I P$ (equivalently, $g_F=-g_I$ in the even sector and $g_F=+g_I$ in the odd sector), which fixes the allowed momentum grid. In each such sector the Hamiltonian can be written in the full-grid form
\begin{align}
    H_{\mathrm{spin}}^{g_F} = \frac{1}{2}\sum_{k\in\mathrm{grid}(g_F)} \operatorname{coeff}_k\, h_k,
\end{align}
with $\operatorname{coeff}_k=\epsilon_k$ for generic modes and $\operatorname{coeff}_k=w_k$ (signed) for the special modes (when present). The chosen-operator vacuum, defined by
\begin{align}
    \hat{a}_k\ket{0} = 0 \quad \text{for all } k,
\end{align}
has $\langle h_k\rangle=-1$ for all modes (with the special modes unoccupied), and therefore its energy is
\begin{align}
    E_{\mathrm{vac}}^{g_F} = -\frac{1}{2}\sum_{k\in\mathrm{grid}(g_F)} \operatorname{coeff}_k.
\end{align}
For $g_F=-1$ (half-integer $k$ grid) there are no special modes and this reduces to $E_{\mathrm{vac}}^{g_F=-1}=-\sum_{k=1/2}^{(N-1)/2}\epsilon_k$. For $g_F=+1$ (integer $k$ grid) the special modes contribute $\operatorname{coeff}_0=w_0$ and $\operatorname{coeff}_{N/2}=w_{N/2}$ (signed). Thus, if a special-mode coefficient is negative, the parity-unconstrained energy-minimizing assignment has $\langle h_k\rangle=+1$ for that mode, not the vacuum value $-1$. In a fixed parity sector the true sector ground state can differ from both the chosen-operator vacuum and the parity-unconstrained BdG reference when the allowed fermion parity requires or forbids occupying one or more modes; this distinction is tested against exact diagonalization in the code.

Let $E_0^{\mathrm{even}}$ and $E_0^{\mathrm{odd}}$ denote the lowest energies in the even and odd parity sectors, respectively.
There are two distinct energy gaps:
\begin{align}
    \Delta_{\mathrm{intra}} &= 2\epsilon_{\min},\\
    \Delta_{\mathrm{inter}} &= |E_0^{\mathrm{odd}} - E_0^{\mathrm{even}}|.
\end{align}
Here $\Delta_{\mathrm{intra}}$ is the parity-preserving excitation gap within a sector, $\Delta_{\mathrm{inter}}$ is the parity-changing gap, and $\epsilon_{\min}$ is the smallest generic mode energy $\epsilon_k$ on the corresponding momentum grid.

The hierarchy $\Delta_{\mathrm{inter}} < \Delta_{\mathrm{intra}}$ arises when the energy difference between filled modes in different parity sectors is smaller than creating a particle-hole pair within a sector. For parity-preserving dynamics, excitations are confined within sectors, making $\Delta_{\mathrm{intra}}$ the relevant energy scale.

This distinction is important for the tensor-network cooling diagnostics.  The
stored array \texttt{mode\_ek\_values} contains the single-quasiparticle
energies $\epsilon_k$ in code units.  By contrast, the automatic bath-detuning
reference for a parity-preserving local system coupling is the many-body scale
$\Delta_{\mathrm{intra}}=2\epsilon_{\min}$, because such a coupling can change
the generic Bogoliubov occupations only in pairs within a fixed parity sector.
Thus one should not use the minimum of \texttt{mode\_ek\_values} itself as the
default detuning for the parity-preserving Ising cooling channel.  On a
Fourier grid containing special modes, lower fixed-parity transitions involving
those special modes may occur; such cases should use an explicit detuning
interval until the special-mode transition scale is selected separately.



\subsection{Parity-preserving couplings in the code convention}
In the notes basis the Ising parity is
\begin{align}
    P=\prod_{n=1}^N \sigma_{n,z},
\end{align}
and it commutes with the spin Hamiltonian \(H_{\mathrm{spin}}\).  In the Julia code the
Ising Hamiltonian is written in the globally rotated basis
\(H_{\mathrm{code}}=J\sum_j\sigma^z_j\sigma^z_{j+1}
+h\sum_j\sigma^x_j\), so this same parity is represented by
\begin{align}
    P_x=\prod_{n=1}^N \sigma_{n,x}.
\end{align}
Thus a local code-basis system Pauli preserves the Ising parity precisely when
it is \(\sigma_x\).

The implemented system-bath coupling is not encoded by a separate fixed bath
example.  It is specified by the shared two-letter coupling label
\(c\in\{X,Y,Z\}^2\), whose source of truth is
\texttt{src/coupling\_utils.jl}.  The local terms are
\begin{align}
    \mathcal T(AA)&=\{(A,A)\},\qquad
    \mathcal T(AB)=\{(A,B),(B,A)\}\quad(A\ne B),\\
    V_{SB}(c)&=
    g\sum_{i=1}^N\sum_{(\mu,\nu)\in\mathcal T(c)}
    \sigma^{\mu}_{s_i,\mathrm{code}}\sigma^{\nu}_{b_i}.
\end{align}
The bath Hamiltonian and reset state use the same convention as the ED, MPS,
and MPO implementations:
\begin{align}
    H_B(c,\Delta)=\frac{\Delta}{2}\sum_{i=1}^N
    \sigma^{\beta(c)}_{b_i},\qquad
    \ket{\Omega_B(c)}=\bigotimes_{i=1}^N
    \ket{\beta(c),-}_{b_i},
\end{align}
where \(\beta(c)\) is the Pauli label returned by
\texttt{get\_bath\_operator} and \(\ket{\beta(c),-}\) is its eigenvalue
\(-1\) state.  This is the convention used by
\texttt{construct\_system\_bath\_hamiltonian} and by the tensor-network note.

Consequently the code-basis coupling \(c\) preserves the Ising parity if and
only if every system-side label \(\mu\) appearing in \(\mathcal T(c)\) is
\(X\).  The large-\(N\) mode-detuning driver checks this condition through
\texttt{ising\_mode\_detuning\_preserves\_px}.  For example, \(XX\)
preserves \(P_x\), while a mixed symmetric label such as \(XY\) or \(XZ\) does
not, because the Hermitian convention also includes a term with a non-\(X\)
system Pauli.  The automatic parity-preserving detuning reference
\(\Delta_{\mathrm{intra}}=2\epsilon_{\min}\) should therefore be used only
when this condition holds; otherwise the physical detuning interval should be
specified explicitly.


\section{Numerical Simulation of Cooling Dynamics}


\subsection{ED and Tensor-Network Time Discretizations}

For a fixed cooling cycle, let
\[
    H_{SB}(c,\Delta)=H_S+H_B(c,\Delta)+V_{SB}(c)
\]
denote the full system-bath Hamiltonian in the interleaved code basis.  In the
exact-diagonalization backend, continuous evolution applies the full
Hamiltonian exponential,
\begin{equation}
    \begin{aligned}
        \ket{\Psi_{SB}(t)}
        &=e^{-iH_{SB}(c,\Delta)t}\ket{\Psi_{SB}(0)},\\
        \rho_{SB}(t)
        &=e^{-iH_{SB}(c,\Delta)t}\rho_{SB}(0)e^{iH_{SB}(c,\Delta)t}.
    \end{aligned}
\end{equation}
When the ED backend is called with the Trotter-evolution option, the parameter
\(\tau\) is an upper bound on the substep size rather than the physical step
itself.  The code chooses
\begin{equation}
    M=\left\lceil \frac{t}{\tau}\right\rceil,\qquad
    \delta t=\frac{t}{M},
    \label{eq:ed_trotter_time_grid}
\end{equation}
up to the small roundoff guard used at exact multiples of \(\tau\), and then
applies the same full-Hamiltonian exponential \(M\) times.  Thus the ED
``Trotter'' path is a time-grid convention for the exact dense or sparse
propagator; it is not the first-order split
\((e^{-iH_z\delta t}e^{-iH_x\delta t})^M\).

The tensor-network Trotter path is different.  It acts on the interleaved chain
\((s_1,b_1,s_2,b_2,\ldots,s_N,b_N)\) and approximates a time slice by a
symmetric product of local gates,
\begin{equation}
    \begin{aligned}
        U_{\delta t}^{\mathrm{TN}}
        &=G_1^{(1/2)}G_2^{(1/2)}\cdots G_m^{(1/2)}
          G_m^{(1/2)}\cdots G_2^{(1/2)}G_1^{(1/2)},\\
        G_j^{(1/2)}
        &=e^{-ih_j\delta t/2}.
    \end{aligned}
\end{equation}
where the \(h_j\) are the two-site \((s_i,b_i)\) local system-bath terms and
the three-site \((s_i,b_i,s_{i+1})\) nearest-neighbor Ising terms described in
the tensor-network note.  This is an interleaved gate decomposition of the
same full system-bath Hamiltonian, not the standard \(H_z/H_x\) split of the
system Hamiltonian alone.  Its errors therefore come from the local
Trotter-Suzuki decomposition and, in MPO or MPS calculations, from the chosen
bond-dimension truncation.


\subsection{Partial Trace for Density Matrices}

In the simulation of the cooling dynamics, we need to trace out bath qubits
from a density matrix \(\rho\) represented in the computational basis.  In the
dense ED implementation the system-bath basis is interleaved,
\[
    \ket{s_1,b_1,s_2,b_2,\ldots,s_N,b_N},
\]
so tracing out the bath keeps the odd sites
\(\{1,3,\ldots,2N-1\}\), while tracing out the system keeps the even sites
\(\{2,4,\ldots,2N\}\).  More generally, if \(K\) is the ordered set of kept
qubits and \(T\) is the ordered set of traced qubits, the reduced matrix is
computed as
\begin{equation}
    (\tr_T \rho)_{ij}
    =
    \sum_{\alpha}
    \rho_{I(i,\alpha;K,T),\, I(j,\alpha;K,T)} ,
    \label{eq:ed_partial_trace_index}
\end{equation}
where \(I(i,\alpha;K,T)\) is the full computational-basis index obtained by
placing the kept bitstring \(i\) on \(K\) and the traced bitstring \(\alpha\)
on \(T\).  This is the map implemented by
\texttt{partial\_trace\_ed} and \texttt{construct\_index}.

For example, tracing out the first qubit corresponds to summing subblocks of
the density matrix.  One partitions \(\rho\) into four blocks according to the
state of that qubit:
\begin{equation}
    \rho = \begin{pmatrix} \rho_{00} & \rho_{01} \\ \rho_{10} & \rho_{11} \end{pmatrix}.
\end{equation}
The traced-out matrix is \(\rho'=\rho_{00}+\rho_{11}\).

In exact arithmetic, the partial trace of a density matrix is Hermitian and
has trace one.  The dense ED code therefore represents the reduced block by
the numerical representative
\begin{equation}
    \rho_{\mathrm{red}}^{\mathrm{num}}
    =
    \frac{\frac12(\rho_{\mathrm{red}}+\rho_{\mathrm{red}}^\dagger)}
         {\tr\!\left[\frac12(\rho_{\mathrm{red}}+\rho_{\mathrm{red}}^\dagger)\right]} .
    \label{eq:ed_reduced_density_canonicalization}
\end{equation}
This Hermitian, trace-one representative removes roundoff-level
anti-Hermitian components introduced by finite-precision arithmetic; it does
not define a different cooling channel.
The bath magnetization recorded on this density-matrix path is the
expectation value
\begin{equation}
    m_B=\frac{1}{N_B}\sum_{i=1}^{N_B}
    \tr\!\left(\rho_B\,\sigma^z_{b_i}\right),
    \label{eq:ed_bath_magnetization_expectation}
\end{equation}
with the computational convention \(0\mapsto +1\), \(1\mapsto -1\).  In the
public result dictionary this expectation-valued diagnostic is stored under
\texttt{bath\_mag\_list}.

\subsection{Sampling for Quantum Trajectories}

For the Monte Carlo wave function method, each cooling cycle involves the following steps:
\begin{enumerate}
    \item Evolve the joint pure state:
          \[
              \ket{\Psi_{SB}'} =
              e^{-iH_{SB}(c,\Delta)t}\bigl(\ket{\Psi_S}\otimes\ket{\Omega_B(c)}\bigr).
          \]
    \item Compute the probability distribution for measuring the bath in the chosen product basis \(\{\ket{n}_B\}\):
          \begin{equation}
              p_n =
              \left\|(\Id_S \otimes {}_B\bra{n})\ket{\Psi_{SB}'}\right\|^2.
          \end{equation}
          This vector of $2^{N_B}$ probabilities can be computed efficiently by taking the diagonal elements of the bath's reduced density matrix, $\rho_B = \tr_S(\ket{\Psi_{SB}'}\bra{\Psi_{SB}'})$.
    \item Sample a specific outcome $\tilde{n}$ from the distribution $\{p_n\}$. A standard way is to compute the cumulative sum of the probabilities, $C_k = \sum_{j=1}^k p_j$, draw a single random number $r \in [0,1]$, and find the first $k$ such that $r \le C_k$.
    \item The new system state is the postmeasurement state, properly normalized:
          \begin{equation}
              \ket{\Psi_S^{\text{new}}} = \frac{(\Id_S \otimes \bra{\tilde{n}}_B) \ket{\Psi_{SB}'}}{\sqrt{p_{\tilde{n}}}}.
          \end{equation}
\end{enumerate}

The corresponding bath diagnostic is not the expectation value in
\cref{eq:ed_bath_magnetization_expectation}; it is the sampled magnetization
of the observed bit string \(\tilde n=(\tilde n_1,\ldots,\tilde n_{N_B})\):
\begin{equation}
    m_B^{\mathrm{sample}}(\tilde n)
    =\frac{1}{N_B}\sum_{i=1}^{N_B} z(\tilde n_i),
    \qquad
    z(0)=+1,\quad z(1)=-1.
    \label{eq:ed_bath_sample_magnetization}
\end{equation}
The ED and tensor-network MCWF implementations store this sampled observable
under \texttt{nb\_list}.  Thus \texttt{bath\_mag\_list} denotes a reduced
density-matrix expectation value, while \texttt{nb\_list} denotes a
postmeasurement sample with the same normalized Pauli-\(Z\) convention.
